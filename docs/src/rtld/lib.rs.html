61    pub fn get_dynstr(&self, offset: usize) -> &str {
62        unsafe {
63            let ptr = self.dynstr.add(offset);
64            let mut len = 0;
65            while *ptr.add(len) != 0 {
66                len += 1;
67            }

69            std::str::from_utf8_unchecked(std::slice::from_raw_parts(ptr, len))
70        }

81    pub fn get_symbol_by_name(&self, name: &str) -> Option<Sym> {
82        let hash = zelf::hash::hash(name.as_bytes());
83
84        let bucket = self.hash_bucket();
85        let chain = self.hash_chain();
86
87        let mut hash_index = bucket[(hash as usize) % bucket.len()];
88        while hash_index != 0 {
89            let sym = unsafe { *self.dynsym.add(hash_index as usize) };
90
91            let is_common =
92                sym.st_shndx.get(LittleEndian) == 0 || sym.st_shndx.get(LittleEndian) == SHN_COMMON;
93
94            if !is_common && self.get_dynstr(sym.st_name.get(LittleEndian) as usize) == name {
95                return Some(sym);
96            }
97
98            hash_index = chain[hash_index as usize];
99        }
100
101        None
102    }

115    fn try_patch_absolute_reloc_impl(
116        &mut self,
117        replacement: *const (),
118        name: &str,
119        entry: Rel,
120    ) -> Option<*const ()> {
121        let ty = entry.r_type(LittleEndian);
122        let sym = entry.r_sym(LittleEndian);
123
124        if ty != R_AARCH64_ABS32 && ty != R_AARCH64_ABS64 && ty != R_AARCH64_GLOB_DAT {
125            return None;
126        }
127
128        let sym = unsafe { *self.dynsym.add(sym as usize) };
129        let symbol_name = self.get_dynstr(sym.st_name.get(LittleEndian) as usize);
130        if symbol_name != name {
131            return None;
132        }
133
134        unsafe {
135            let prev = *(self
136                .module_base
137                .add(entry.r_offset.get(LittleEndian) as usize)
138                as *const *const ());
139            *(self
140                .module_base
141                .add(entry.r_offset.get(LittleEndian) as usize) as *mut *const ()) = replacement;
142            Some(prev)
143        }
144    }

146    fn try_patch_reloc_impl(
147        &mut self,
148        replacement: *const (),
149        name: &str,
150        entry: Rel,
151    ) -> Option<*const ()> {
152        let ty = entry.r_type(LittleEndian);
153        let sym = entry.r_sym(LittleEndian);
154
155        if ty != R_AARCH64_JUMP_SLOT {
156            return None;
157        }
158
159        let sym = unsafe { *self.dynsym.add(sym as usize) };
160        let symbol_name = self.get_dynstr(sym.st_name.get(LittleEndian) as usize);
161        if symbol_name != name {
162            return None;
163        }
164
165        unsafe {
166            let prev = *(self
167                .module_base
168                .add(entry.r_offset.get(LittleEndian) as usize)
169                as *const *const ());
170            *(self
171                .module_base
172                .add(entry.r_offset.get(LittleEndian) as usize) as *mut *const ()) = replacement;
173            Some(prev)
174        }
175    }

177    pub fn try_patch_absolute_reloc(
178        &mut self,
179        replacement: *const (),
180        name: &str,
181    ) -> Option<*const ()> {
182        for index in self.rela_count..(self.rela_dyn_size / std::mem::size_of::<Rela>()) {
183            let entry = unsafe { *self.rela.add(index) };
184            if let Some(prev) = self.try_patch_absolute_reloc_impl(
185                replacement,
186                name,
187                Rel {
188                    r_info: entry.r_info,
189                    r_offset: entry.r_offset,
190                },
191            ) {
192                return Some(prev);
193            }
194        }
195
196        None
197    }

217    pub fn get_module_name(&self) -> Option<&'static str> {
218        let info = nx::query_memory(self.module_base as u64);
219
220        let ro_info = nx::query_memory(info.addr + info.size);
221
222        unsafe {
223            let rw_data_offset = *(ro_info.addr as *const u32);
224            if rw_data_offset as u64 + info.addr == ro_info.addr + ro_info.size {
225                return None;
226            }
227
228            if rw_data_offset != 0 {
229                return None;
230            }
231
232            let path_length = *(ro_info.addr as *const i32).add(1);
233            if path_length <= 0 {
234                return None;
235            }
236
237            let name = std::str::from_utf8_unchecked(std::slice::from_raw_parts(
238                (ro_info.addr + 8) as *const u8,
239                path_length as usize,
240            ));
241
242            let split = name.split('\\').last().unwrap();
243            let name = split.split('/').last().unwrap();
244
245            Some(name)
246        }
247    }

249    pub fn get_address_range(&self, section: Section) -> Range<u64> {
250        match section {
251            Section::Text => {
252                let info = nx::query_memory(self.module_base as u64);
253                info.addr..info.addr + info.size
254            }
255            Section::RoData => {
256                let text = self.get_address_range(Section::Text);
257                let info = nx::query_memory(text.end);
258
259                info.addr..info.addr + info.size
260            }
261            Section::Data => {
262                let ro_data = self.get_address_range(Section::RoData);
263                let info = nx::query_memory(ro_data.end);
264                info.addr..info.addr + info.size
265            }
266        }
267    }

269    pub fn get_symbol_range_for_address(&self, address: u64) -> Option<Range<u64>> {
270        let symbols =
271            unsafe { std::slice::from_raw_parts(self.dynsym, self.hash_nchain_value as usize) };
272
273        for symbol in symbols {
274            let shndx = symbol.st_shndx.get(LittleEndian);
275            if shndx == 0 || (shndx & 0xFF00) == 0xFF00 {
276                continue;
277            }
278
279            if symbol.st_info & 0xF != 2 {
280                continue;
281            }
282
283            let function_start = unsafe {
284                self.module_base
285                    .add(symbol.st_value.get(LittleEndian) as usize) as u64
286            };
287
288            let function_end = function_start + symbol.st_size.get(LittleEndian);
289
290            if function_start <= address && address < function_end {
291                return Some(function_start..function_end);
292            }
293        }
294
295        None
296    }

314pub struct ModuleObjectListIterator {
315    end: *mut ModuleObject,
316    current: *mut ModuleObject,
317}
318
319impl Iterator for ModuleObjectListIterator {
320    type Item = &'static mut ModuleObject;
321
322    fn next(&mut self) -> Option<Self::Item> {
323        if self.current == self.end {
324            None
325        } else {
326            let next = unsafe { (*self.current).next };
327            let current = unsafe { &mut *self.current };
328            self.current = next;
329            Some(current)
330        }
331    }

372pub fn find_module_for_address(
373    address: u64,
374    section: Section,
375) -> Option<&'static mut ModuleObject> {
376    unsafe {
377        AUTO_LOAD_LIST
378            .iter()
379            .chain(MANUAL_LOAD_LIST.iter())
380            .find(|object| {
381                let range = object.get_address_range(section);
382                range.contains(&address)
383            })
384    }
385}
386
387pub fn find_module_for_address_no_section(address: u64) -> Option<&'static mut ModuleObject> {
388    unsafe {
389        AUTO_LOAD_LIST
390            .iter()
391            .chain(MANUAL_LOAD_LIST.iter())
392            .find(|object| {
393                object.get_address_range(Section::Text).contains(&address)
394                    || object.get_address_range(Section::Data).contains(&address)
395                    || object.get_address_range(Section::RoData).contains(&address)
396            })
397    }
398}

400pub fn find_module_by_name(name: &str) -> Option<&'static mut ModuleObject> {
401    let mut objects = unsafe { AUTO_LOAD_LIST.iter().chain(MANUAL_LOAD_LIST.iter()) };
402
403    objects.find(|object| object.get_module_name().unwrap_or("__invalid_name") == name)
404}

406pub fn is_valid_pointer_for_section(address: u64, section: Section) -> bool {
407    find_module_for_address(address, section).is_some()

410#[derive(Debug, Copy, Clone, PartialEq, Eq)]
411pub enum MemoryState {
412    NonDereferencable,
413    DereferencableOutsideModule,
414    DereferencableInModule,
415}
416
417pub fn get_memory_state(address: u64) -> MemoryState {
418    let state = nx::query_memory(address);
419    if state.perm & 0x1 == 0 {
420        return MemoryState::NonDereferencable;
421    }
422
423    if find_module_for_address_no_section(address).is_some() {
424        MemoryState::DereferencableInModule
425    } else {
426        MemoryState::DereferencableOutsideModule
427    }
