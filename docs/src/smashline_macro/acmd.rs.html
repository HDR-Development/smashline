7mod kw {
8    syn::custom_keyword!(agent);
9    syn::custom_keyword!(script);
10    syn::custom_keyword!(scripts);
11    syn::custom_keyword!(category);
12    syn::custom_keyword!(low_priority);
13    syn::custom_keyword!(high_priority);
14}

16enum AcmdAttribute {
17    Agent(syn::Expr),
18    Script(syn::Expr),
19    Scripts(Vec<syn::Expr>),
20    Category(syn::Expr),
21    LowPriority,
22    HighPriority,
23}
24
25impl Parse for AcmdAttribute {
26    fn parse(input: syn::parse::ParseStream) -> syn::Result<Self> {
27        crate::match_kw! {
28            input;
29            kw::agent [=] => input.parse().map(Self::Agent),
30            kw::script [=] => input.parse().map(Self::Script),
31            kw::scripts [=] => {
32                let bracketed;
33                syn::bracketed!(bracketed in input);
34                Punctuated::<_, syn::token::Comma>::parse_terminated(&bracketed).map(|p| Self::Scripts(p.into_iter().collect()))
35            },
36            kw::category [=] => input.parse().map(Self::Category),
37            kw::low_priority => Ok(Self::LowPriority),
38            kw::high_priority => Ok(Self::HighPriority);
39            _ => Err(syn::Error::new(input.span(), "unsupported item"))
40        }
41    }
42}

44pub struct AcmdAttributes {

46    scripts: Vec<syn::Expr>,
47    category: syn::Expr,
48    priority: syn::Expr,

51fn name_to_cat(name: &str) -> Option<syn::Expr> {
52    if name.starts_with("game_") {
53        Some(syn::parse_quote!(::smashline::Acmd::Game))
54    } else if name.starts_with("effect_") {
55        Some(syn::parse_quote!(::smashline::Acmd::Effect))
56    } else if name.starts_with("sound_") {
57        Some(syn::parse_quote!(::smashline::Acmd::Sound))
58    } else if name.starts_with("expression_") {
59        Some(syn::parse_quote!(::smashline::Acmd::Expression))
60    } else {
61        None
62    }
63}
64
65fn literals_to_cat(exprs: &[syn::Expr]) -> Option<syn::Expr> {
66    for expr in exprs.iter() {
67        match expr {
68            syn::Expr::Lit(syn::ExprLit {
69                lit: syn::Lit::Str(lit_str),
70                ..
71            }) => {
72                if let Some(cat) = name_to_cat(lit_str.value().as_str()) {
73                    return Some(cat);
74                }
75            }
76            _ => {}
77        }
78    }
79
80    None
81}

83impl AcmdAttributes {
84    pub fn installer(&self, function: &syn::Ident, crate_tokens: TokenStream) -> TokenStream {
85        let Self {
86            agent,
87            category,
88            priority,
89            scripts,
90        } = self;
91
92        quote::quote! {
93            pub fn install() {
94                #(
95                    #crate_tokens::api::install_acmd_script(
96                        #crate_tokens::AsHash40::as_hash40(#agent),
97                        #crate_tokens::AsHash40::as_hash40(#scripts),
98                        #category,
99                        #priority,
100                        #function
101                    );
102                )*
103            }
104        }
105    }
106
107    fn parse_named(input: ParseStream) -> syn::Result<Self> {
108        let attrs = Punctuated::<AcmdAttribute, syn::token::Comma>::parse_terminated(input)?;
109
110        let mut agent = None;
111        let mut scripts = vec![];
112        let mut category = None;
113        let mut priority = None;
114        for attr in attrs {
115            match attr {
116                AcmdAttribute::Agent(expr) => agent = Some(expr),
117                AcmdAttribute::Script(expr) => scripts.push(expr),
118                AcmdAttribute::Scripts(exprs) => scripts.extend(exprs),
119                AcmdAttribute::Category(expr) => category = Some(expr),
120                AcmdAttribute::LowPriority => {
121                    priority = Some(syn::parse_quote!(::smashline::Priority::Low))
122                }
123                AcmdAttribute::HighPriority => {
124                    priority = Some(syn::parse_quote!(::smashline::Priority::High))
125                }
126            }
127        }
128
129        let Some(agent) = agent else {
130            return Err(syn::Error::new(input.span(), "agent must be provided"));
131        };
132
133        if scripts.is_empty() {
134            return Err(syn::Error::new(
135                input.span(),
136                "at least 1 script must be provided",
137            ));
138        }
139
140        let priority =
141            priority.unwrap_or_else(|| syn::parse_quote!(::smashline::Priority::Default));
142
143        category = category.or_else(|| literals_to_cat(&scripts));
144
145        let Some(category) = category else {
146            return Err(syn::Error::new(
147                input.span(),
148                "could not infer acmd category from scripts"
149            ));
150        };
151
152        Ok(Self {
153            agent,
154            scripts,
155            category,
156            priority,
157        })
158    }
159
160    fn parse_unnamed(input: ParseStream) -> syn::Result<Self> {
161        let agent: syn::Expr = input.parse()?;
162        let _: syn::Token![,] = input.parse()?;
163        let scripts: Vec<_> = if input.peek(syn::token::Bracket) {
164            let bracketed;
165            syn::bracketed!(bracketed in input);
166
167            let scripts = Punctuated::<syn::Expr, syn::token::Comma>::parse_terminated(&bracketed)?;
168            scripts.into_iter().collect()
169        } else {
170            let script: syn::Expr = input.parse()?;
171            vec![script]
172        };
173
174        let Some(category) = literals_to_cat(&scripts) else {
175            return Err(syn::Error::new(
176                input.span(),
177                "could not infer acmd category from scripts, specify it manually",
178            ));
179        };
180
181        Ok(Self {
182            agent,
183            scripts,
184            category,
185            priority: syn::parse_quote!(::smashline::Priority::Default),
186        })
187    }
188}
189
190impl Parse for AcmdAttributes {
191    fn parse(input: ParseStream) -> syn::Result<Self> {
192        Self::parse_named(input).or_else(|_| Self::parse_unnamed(input))
193    }
194}
