42fn map_to_bare_fn_arg(arg: &syn::FnArg) -> syn::Result<syn::BareFnArg> {
43    match arg {
44        syn::FnArg::Typed(typed) => Ok(syn::BareFnArg {
45            attrs: typed.attrs.clone(),
46            name: None,
47            ty: (*typed.ty).clone(),
48        }),
49        syn::FnArg::Receiver(receiver) => {
50            Err(syn::Error::new(receiver.span(), "receivers not supported"))
51        }
52    }
53}

55pub fn map_to_bare_fn(sig: &syn::Signature) -> syn::Result<syn::TypeBareFn> {
56    let abi = syn::Abi {
57        extern_token: syn::token::Extern(sig.fn_token.span()),
58        name: Some(syn::LitStr::new("C", sig.fn_token.span())),
59    };
60
61    let mut inputs = Punctuated::new();
62    for pair in sig.inputs.pairs() {
63        let value = match pair {
64            Pair::Punctuated(value, _) => map_to_bare_fn_arg(value)?,
65            Pair::End(value) => map_to_bare_fn_arg(value)?,
66        };
67
68        inputs.push(value);
69    }
70
71    Ok(syn::TypeBareFn {
72        lifetimes: None,
73        unsafety: None,
74        abi: Some(abi),
75        fn_token: sig.fn_token.clone(),
76        paren_token: sig.paren_token.clone(),
77        inputs,
78        // TODO: Add support for variadics
79        variadic: None,
80        output: sig.output.clone(),
81    })
82}

84pub fn extract_args(sig: &syn::Signature) -> syn::Result<Vec<syn::Ident>> {
85    let mut args = vec![];
86
87    for input in sig.inputs.iter() {
88        match input {
89            syn::FnArg::Typed(syn::PatType { pat, .. }) => {
90                let syn::Pat::Ident(syn::PatIdent { ident, .. }) = &**pat else {
91                    return Err(syn::Error::new(pat.span(), "expected identifier pattern"));
92                };
93                args.push(ident.clone());
94            }
95            other => {
96                return Err(syn::Error::new(other.span(), "expected identifier pattern"));
97            }
98        }
99    }
100
101    Ok(args)
102}
