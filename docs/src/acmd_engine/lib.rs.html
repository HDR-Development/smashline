1use action::{ActionError, ActionRegistry, DynamicAction};
2use asset::{Category, SmashlineScriptV0, VersionedSmashlineScript};
3use hash40::Hash40;
4use serde_json::Value;
5
6pub mod action;
7pub mod asset;
8
9pub struct SmashlineScript {
10    pub agent: Hash40,
11    pub category: Category,
12    pub replace: Hash40,
13    pub actions: Vec<DynamicAction>,
14}
15
16impl SmashlineScript {
17    fn from_v0(registry: &ActionRegistry, script: SmashlineScriptV0) -> Result<Self, ActionError> {
18        Ok(Self {
19            agent: script.agent,
20            category: script.category,
21            replace: script.replace,
22            actions: script
23                .actions
24                .into_iter()
25                .map(|action| registry.as_action(action))
26                .collect::<Result<Vec<_>, _>>()?,
27        })
28    }
29
30    pub fn from_json(
31        registry: &ActionRegistry,
32        json: impl AsRef<str>,
33    ) -> Result<Self, ActionError> {
34        let versioned: VersionedSmashlineScript =
35            serde_json::from_str(json.as_ref()).map_err(ActionError::ParseError)?;
36
37        match versioned {
38            VersionedSmashlineScript::V0(v0) => Self::from_v0(registry, v0),
39        }
40    }
41
42    pub fn to_json(&self, registry: &ActionRegistry) -> Result<String, ActionError> {
43        let versioned = VersionedSmashlineScript::from_latest(registry, self)?;
44        serde_json::to_string_pretty(&versioned).map_err(ActionError::SerializeError)
45    }
46
47    pub fn to_json_value(&self, registry: &ActionRegistry) -> Result<Value, ActionError> {
48        let versioned = VersionedSmashlineScript::from_latest(registry, self)?;
49        serde_json::to_value(versioned).map_err(ActionError::SerializeError)
50    }
51}
