60            Init => Self::Init(unsafe { std::mem::transmute(function) }),

62            ExecStop => Self::ExecStop(unsafe { std::mem::transmute(function) }),
63            Post => Self::Post(unsafe { std::mem::transmute(function) }),
64            Exit => Self::Exit(unsafe { std::mem::transmute(function) }),
65            MapCorrection => Self::MapCorrection(unsafe { std::mem::transmute(function) }),
66            FixCamera => Self::FixCamera(unsafe { std::mem::transmute(function) }),
67            FixPosSlow => Self::FixPosSlow(unsafe { std::mem::transmute(function) }),
68            CheckDamage => Self::CheckDamage(unsafe { std::mem::transmute(function) }),
69            CheckAttack => Self::CheckAttack(unsafe { std::mem::transmute(function) }),

94#[inline(always)]
95fn call_callback1(fighter: &mut L2CFighterBase, callback: Callback1, ctx: &InlineCtx) {
96    let arg = unsafe { std::mem::transmute(ctx.registers[1].x()) };
97    callback(fighter, arg);
98}

100#[inline(always)]
101fn call_callback2(fighter: &mut L2CFighterBase, callback: Callback2, ctx: &InlineCtx) {
102    let arg = unsafe { std::mem::transmute(ctx.registers[2].x()) };
103    let arg2 = unsafe { std::mem::transmute(ctx.registers[3].x()) };
104    callback(fighter, arg, arg2);
105}

107macro_rules! decl_functions {
108    ($($name:ident($line:ident, $offset:expr, $code_cave:expr, $reg:expr) => $call_fn:ident);*) => {
109        $(
110            extern "C" fn $name(ctx: &InlineCtx) {
111                let fighter: &'static mut L2CFighterBase =
112                    unsafe { std::mem::transmute(ctx.registers[$reg].x()) };
113                    
114                let fighter = std::hint::black_box(fighter);
115
116                let callbacks = crate::create_agent::status_callbacks(fighter);
117
118                for callback in callbacks.iter() {
119                    if let StatusCallbackFunction::$line(callback_fn) = *callback {
120                       $call_fn(fighter, callback_fn, ctx);
121                    }
122                }
123            }
124        )*
125
126        pub fn install_callback_hooks() {
127            $(
128                crate::nro_hook::add_hook($offset, $code_cave, $name);
129            )*
130            unsafe {
131                crate::nro_hook::add_normal_hook(0x3250, &mut ORIGINAL, call_line_status_hook as *const ());
132            }
133        }
134    }
135}

165static mut ORIGINAL: *const () = std::ptr::null();

167unsafe fn call_line_status_hook(
168    fighter: &mut L2CFighterBase,
169    variadic: &mut Variadic,
170    string: *const u8,
171    va_list: u32,
172) {
173    let callable: extern "C" fn(&mut L2CFighterBase, &mut Variadic, *const u8, u32) =
174        std::mem::transmute(ORIGINAL);
175    callable(fighter, variadic, string, va_list);
176
177    let callbacks = crate::create_agent::status_callbacks(fighter);
178    for callback in callbacks.iter() {
179        if let StatusCallbackFunction::Main(callback_fn) = *callback {
180            callback_fn(fighter);
181        }
182    }
