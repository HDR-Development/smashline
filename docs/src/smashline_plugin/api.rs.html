22fn mark_costume(
23    agent: Hash40,
24    costume: Costume,
25) {
26    let costume_slice = costume.as_slice();
27
28    if costume_slice.is_empty() {
29        return;
30    }
31
32    let mut costumes = crate::create_agent::COSTUMES.write();
33    let costumes = costumes
34        .entry(agent)
35        .or_default();
36
37    for c in &mut *costumes {
38        if costume_slice == c.as_slice() {
39            return;
40        }
41
42        let exists = costume_slice.iter().any(|e| {
43            c.as_slice().contains(e)
44        });
45
46        if exists {
47            // TODO: Do something if a costume has already been marked for this agent
48        }
49    }
50
51    costumes.push(costume);
52}

54#[no_mangle]
55pub extern "C" fn smashline_remove_by_plugin_range(start: usize, end: usize) {
56    crate::create_agent::ACMD_SCRIPTS_DEV
57        .write()
58        .clear();
59
60    crate::create_agent::STATUS_SCRIPTS_DEV
61        .write()
62        .clear();
63
64    {
65        let mut callbacks = crate::callbacks::CALLBACKS.write();
66
67        let working = std::mem::take(&mut *callbacks);
68        *callbacks = working
69            .into_iter()
70            .filter(|cb| !(start..end).contains(&cb.function.as_address()))
71            .collect();
72    }
73
74    {
75        let mut callbacks = crate::state_callback::STATE_CALLBACKS.write();
76        let working = std::mem::take(&mut *callbacks);
77        *callbacks = working
78            .into_iter()
79            .filter(|cb| !(start..end).contains(&(cb.function as *const () as usize)))
80            .collect();
81    }
82}

303#[no_mangle]
304pub extern "C" fn smashline_clone_weapon(
305    original_owner: StringFFI,
306    original_article_id: i32,
307    new_owner: StringFFI,
308    new_name: StringFFI,
309    use_original_code: bool,
310) -> i32 {
311    let original_owner = original_owner.as_str().unwrap().to_string();
312    let new_owner = new_owner.as_str().unwrap().to_string();
313    let new_name = new_name.as_str().unwrap().to_string();
314
315    let original_owner_id = LOWERCASE_FIGHTER_NAMES
316        .iter()
317        .position(|name| name == original_owner)
318        .unwrap();
319
320    // let original_name_id = LOWERCASE_WEAPON_NAMES
321    //     .iter()
322    //     .position(|name| name == original_name)
323    //     .unwrap();
324
325    let original_name = LOWERCASE_WEAPON_NAMES.get(original_article_id as usize).unwrap();
326
327    let new_owner_id = LOWERCASE_FIGHTER_NAMES
328        .iter()
329        .position(|name| name == new_owner)
330        .unwrap();
331
332    let mut new_agents = crate::cloning::weapons::NEW_AGENTS.write();
333
334    let mut new_articles = crate::cloning::weapons::NEW_ARTICLES.write();
335    let articles = new_articles
336        .entry(new_owner_id as i32)
337        .or_default();
338
339    if let Some(id) = articles.iter().position(|article|
340        article.original_owner == original_owner_id as i32 &&
341        article.weapon_id == original_article_id
342    ) {
343        return id as i32;
344    }
345
346    for agents in new_agents.values() {
347        if let Some(agent) = agents.iter().find(|agent| 
348            agent.owner_name == new_owner && agent.new_name == new_name
349        ) {
350            let owner = LOWERCASE_FIGHTER_NAMES.get(agent.old_owner_id as usize).unwrap();
351            panic!(
352                "Weapon with the name '{}_{}' has already been cloned, but using '{}_{}' instead of '{}_{}'", 
353                new_owner, new_name, owner, agent.old_name, original_owner, original_name
354            );
355        }
356    }
357
358    new_agents
359        .entry(original_article_id as i32)
360        .or_default()
361        .push(NewAgent {
362            old_owner_id: original_owner_id as i32,
363            owner_id: new_owner_id as i32,
364            owner_name_ffi: format!("{new_owner}\0"),
365            new_name_ffi: format!("{new_name}\0"),
366            owner_name: new_owner,
367            new_name,
368            old_name: original_name.to_string(),
369            use_original_code,
370        });
371
372    let id = articles.len();
373    articles.push(NewArticle {
374        original_owner: original_owner_id as i32,
375        weapon_id: original_article_id,
376    });
377
378    id as i32
379}

392#[no_mangle]
393pub extern "C" fn smashline_whitelist_kirby_copy_article(
394    fighter_id: i32,
395    article_id: i32
396) {
397    let mut copy_whitelist = crate::cloning::weapons::KIRBY_COPY_ARTICLE_WHITELIST.write();
398    if let Some(whitelist) = copy_whitelist.get_mut(&fighter_id) {
399        if whitelist.contains(&article_id) {
400            println!("Copy Whitelist already contains fighter {:#x} article {:#x}!", fighter_id, article_id);
401        }
402        else {
403            whitelist.push(article_id);
404        }
405    }
406    else {
407        copy_whitelist.insert(fighter_id, vec![article_id]);
408    }
