12pub fn get_weapon_name(id: i32) -> Option<String> {
13    let current_owner = CURRENT_OWNER_KIND.load(Ordering::Relaxed);
14    let agents = NEW_AGENTS.read();
15    if let Some(name) = try_get_new_agent(&agents, id, current_owner).map(|agent| agent.new_name.clone()) {
16        Some(name)
17    } else {
18        LOWERCASE_WEAPON_NAMES.get(id as usize).map(|x| x.to_string())
19    }
20}

40pub fn get_costume_from_entry_id(entry_id: i32) -> Option<i32> {
41    unsafe {
42        let text = skyline::hooks::getRegionAddress(skyline::hooks::Region::Text) as *const u64;
43        let some_vec = *text.add(0x5323680 / 0x8);
44
45        let index = entry_id as u64 * 8;
46        let some_struct = *((some_vec + index + 0xe8) as *const u64);
47
48        const COSTUME_OFFSET: u64 = 100;
49        let ptr = (some_struct + COSTUME_OFFSET) as *const i32;
50        if ptr as u64 == 0x64 { // entry articles
51            None
52        } else {
53            Some(*ptr)
54        }
55    }
56}

67pub fn has_costume(hash: Hash40, costume: i32) -> bool {
68    COSTUMES
69        .read()
70        .get(&hash).map_or(false, |costume_vec| {
71            costume_vec.iter().any(|c| {
72                c.as_slice().contains(&(costume as usize))
73            })
74        })
75}

77pub fn get_costume_data(hash: Hash40, costume: i32) -> Costume {
78    let def = Costume::default();
79    COSTUMES
80        .read()
81        .get(&hash).map_or(def, |costume_vec| {
82            costume_vec.iter().find(|c| {
83                c.as_slice().contains(&(costume as usize))
84            }).copied().unwrap_or(def)
85        })
86}

117/// Loads a fighter module via the game's internal methods
118pub fn load_fighter_module(kind: i32) {
119    if kind < 0 {
120        return;
121    }
122
123    // Step 1: Get manager 
124    let manager = dynamic_module_manager();
125
126    // Step 2: Get fighter name
127    let Some(name) = LOWERCASE_FIGHTER_NAMES.get(kind as usize) else {
128        return;
129    };
130
131    // Step 2: Insert the module or inc ref count
132    let module = unsafe {
133        let tree = manager.add(0x38 / 8) as *mut smash::cpp::Tree<Hash40, *mut skyline::nn::ro::Module>;
134
135        if let Some(module) = (*tree).get_mut(&Hash40::new(name)) {
136            *(*module as *mut u64 as *mut i32).add(0x134 / 4) += 1;
137            return;
138        }
139
140        load_file(format!("prebuilt:/nro/release/lua2cpp_{name}.nro"));
141
142        let mut module_name = [0u8; 0x100];
143        module_name[..name.len()].copy_from_slice(name.as_bytes());
144
145        let module = Box::leak(Box::new(skyline::nn::ro::Module {
146            ModuleObject: std::ptr::null_mut(),
147            State: 0,
148            NroPtr: std::ptr::null_mut(),
149            BssPtr: std::ptr::null_mut(),
150            _x20: std::ptr::null_mut(),
151            SourceBuffer: std::ptr::null_mut(),
152            Name: module_name,
153            _x130: 0,
154            _x131: 0,
155            isLoaded: false
156        })) as *mut _;
157
158        // Safe bc Module is aligned to 0x8 and so the size is 0x138
159        *(module as *mut u32).add(0x134 / 4) = 1;
160
161        (*tree).insert(Hash40::new(name), module);
162        module
163    };
164
165    // Step 3: Send command to manager that we want to load a module
166    unsafe {
167        let deque = &mut *(manager.add(0x80 / 8) as *mut MyDeque);
168        let distance = if deque.end.offset_from(deque.start) != 0 {
169            deque.end.offset_from(deque.start) * 0x100 - 1
170        } else {
171            0
172        };
173
174        let next_index = deque.start_index + deque.len;
175        if (deque.start_index + deque.len) as isize == distance {
176            extend_deque(deque);
177        }
178
179        *(*deque.start.add(next_index / 0x100)).add(next_index & 0xFF) = Command {
180            id: 3,
181            arg: module as u64
182        };
183
184        deque.len += 1;
185    }
186
187    // Step 4: Signal event
188    unsafe {
189        signal_event(**(manager.add(0x50 / 8) as *const *const u64));
190    }
191}

196pub fn unload_fighter_module(id: i32) {
197    if id < 0 {
198        return;
199    }
200
201    let Some(name) = LOWERCASE_FIGHTER_NAMES.get(id as usize) else {
202        return;
203    };
204
205    unsafe {
206        dynamic_module_manager_unload(dynamic_module_manager(), &Hash40::new(name));
207    }

210pub fn is_fighter_module_loaded(id: i32) -> bool {
211    if id < 0 {
212        return false;
213    }
214
215    let Some(name) = LOWERCASE_FIGHTER_NAMES.get(id as usize) else {
216        return false;
217    };
218
219    let manager = dynamic_module_manager();
220    unsafe {
221        let tree = manager.add(0x38 / 8) as *const smash::cpp::Tree<Hash40, *mut skyline::nn::ro::Module>;
222
223        (*tree).get(&Hash40::new(name)).map(|x| (**x).isLoaded).unwrap_or_default()
224    }
225}

241pub fn load_file(name: impl Into<String>) {
242    let mut search_path = 0u32;
243    unsafe {
244        get_search_path_index(&mut search_path, format!("{}\0", name.into()).as_ptr());
245    }
246
247    let file_path = unsafe { get_file_path_from_search_path(search_path) };
248    if file_path != 0xFFFFFF {
249        unsafe {
250            add_to_res_service(get_filesystem(), file_path);
251        }
252    }
