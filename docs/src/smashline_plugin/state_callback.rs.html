1use locks::RwLock;
2use skyline::hooks::InlineCtx;
3use smash::app::BattleObject;
4use smashline::{BattleObjectCategory, Costume, Hash40, L2CFighterBase, ObjectEvent};
5
6pub type StateCallbackFunction = unsafe extern "C" fn(&mut L2CFighterBase);
7
8pub struct StateCallback {
9    pub agent: Option<Hash40>,
10    pub event: ObjectEvent,
11    pub function: StateCallbackFunction,
12    pub costume: Costume,
13}
14
15pub static STATE_CALLBACKS: RwLock<Vec<StateCallback>> = RwLock::new(Vec::new());
16
17fn call_state_callback(agent: &mut L2CFighterBase, event: ObjectEvent) {
18    let hash = crate::create_agent::agent_hash(agent);
19    let callbacks = STATE_CALLBACKS.read();
20
21    let object: &mut BattleObject = unsafe {std::mem::transmute(agent.battle_object)};
22    let category = BattleObjectCategory::from_battle_object_id(object.battle_object_id);
23
24    let costume = crate::utils::get_agent_costume(agent.battle_object as *const BattleObject).unwrap_or(0);
25    let has_costume = crate::utils::has_costume(hash, costume);
26
27    for callback in callbacks.iter().filter(|cb| cb.event == event) {
28        if let Some(required) = callback.agent {
29            if hash != required {
30                match category {
31                    Some(BattleObjectCategory::Fighter) => if required != Hash40::new("fighter") { continue; },
32                    Some(BattleObjectCategory::Weapon) => if required != Hash40::new("weapon") { continue; },
33                    _ => { continue; },
34                }
35            } else {
36                let c = callback.costume.as_slice();
37
38                if has_costume && !c.contains(&(costume as usize)) {
39                    continue;
40                }
41
42                if !has_costume && !c.is_empty() {
43                    continue;
44                }
45            }
46        }
47
48        unsafe {
49            (callback.function)(agent);
50        }
51    }
52}

54pub static mut CAN_RUN_ON_START : bool = false;
55
56#[skyline::hook(offset = 0x48ad04, inline)]
57unsafe fn lua_module_start_lua2cpp(_ctx: &InlineCtx) {
58    CAN_RUN_ON_START = true;
59}
60
61#[skyline::hook(offset = 0x48ada0)]
62unsafe fn lua_module_end(lua_module: *const u64) {
63    let agent = std::mem::transmute(*lua_module.add(0x1d8 / 8));
64    call_state_callback(agent, ObjectEvent::End);
65}
66
67#[skyline::hook(offset = 0x48abac, inline)]
68unsafe fn lua_module_initialize_lua2cpp(ctx: &InlineCtx) {
69    let agent = std::mem::transmute(ctx.registers[0].x());
70    call_state_callback(agent, ObjectEvent::Initialize);
71}
72
73#[skyline::hook(offset = 0x48ac44, inline)]
74unsafe fn lua_module_finalize_lua2cpp(ctx: &InlineCtx) {
75    let agent = std::mem::transmute(ctx.registers[0].x());
76    call_state_callback(agent, ObjectEvent::Finalize);
77}
78
79#[skyline::hook(offset = 0x3afde0, inline)]
80unsafe fn start_module_accessor_end(ctx: &mut InlineCtx) {
81    if CAN_RUN_ON_START {
82        let boma = ctx.registers[19].x();
83        let lua_module = *(boma as *mut u64).add(0x190 / 8);
84        let agent = std::mem::transmute(*((lua_module + 0x1D8) as *mut *mut L2CFighterBase));
85        call_state_callback(agent, ObjectEvent::Start);
86    }
87    CAN_RUN_ON_START = false;
88}

90pub fn install_state_callback_hooks() {
