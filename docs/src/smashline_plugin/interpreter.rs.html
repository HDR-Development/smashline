31pub fn get_scripts(fighter_name: &str, weapon_name: Option<&str>) -> Vec<LoadedScript> {
32    let sub_folder = weapon_name.unwrap_or("body");
33    let path = PathBuf::from(format!("mods:/fighter/{fighter_name}/acmd/{sub_folder}/"));
34    let read_dir = match std::fs::read_dir(&path) {
35        Ok(read_dir) => read_dir,
36        Err(e) => {
37            // println!("Failed to get scripts: {e}");
38            return vec![];
39        }
40    };
41
42    let mut scripts = vec![];
43
44    for entry in read_dir {
45        let entry = match entry {
46            Ok(entry) => entry,
47            Err(e) => {
48                println!("Failed to get script entry: {e}");
49                continue;
50            }
51        };
52
53        match entry.file_type() {
54            Ok(ty) if ty.is_file() => {}
55            _ => continue,
56        }
57
58        let Some("acmd") = entry.path().extension().and_then(|s| s.to_str()) else {
59            continue;
60        };
61
62        let data = match std::fs::read_to_string(entry.path()) {
63            Ok(data) => data,
64            Err(e) => {
65                println!("Failed to read {}: {e}", entry.path().display());
66                continue;
67            }
68        };
69
70        let script = match SmashlineScript::from_json(&ACTION_REGISTRY, data) {
71            Ok(data) => data,
72            Err(e) => {
73                println!("Failed to parse {}: {e}", entry.path().display());
74                continue;
75            }
76        };
77
78        scripts.push(LoadedScript {
79            file_name: entry
80                .path()
81                .file_name()
82                .unwrap()
83                .to_string_lossy()
84                .to_string(),
85            script: Arc::new(locks::RwLock::new(Arc::new(script))),
86        });
87    }
88
89    scripts
90}

92pub fn load_single_script(fighter: &str, weapon_name: Option<&str>, file_name: &str) {
93    let sub_folder = weapon_name.unwrap_or("body");
94    let path = PathBuf::from(format!(
95        "mods:/fighter/{fighter}/acmd/{sub_folder}/{file_name}"
96    ));
97    let data = match std::fs::read_to_string(&path) {
98        Ok(data) => data,
99        Err(e) => {
100            println!("Failed to read {}: {e}", path.display());
101            return;
102        }
103    };
104
105    let new_script = match SmashlineScript::from_json(&ACTION_REGISTRY, data) {
106        Ok(data) => data,
107        Err(e) => {
108            println!("Failed to parse {}: {e}", path.display());
109            return;
110        }
111    };
112
113    let agent = if let Some(weapon) = weapon_name {
114        Hash40::new(fighter).concat_str("_").concat_str(weapon)
115    } else {
116        Hash40::new(fighter)
117    };
118
119    let loaded_scripts = LOADED_SCRIPTS.write();
120    let Some(loaded) = loaded_scripts.get(&agent).and_then(|weak| weak.upgrade()) else {
121        println!("No loaded scripts to insert into");
122        return;
123    };
124
125    for script in loaded.iter() {
126        if script.file_name == file_name {
127            *script.script.write() = Arc::new(new_script);
128            return;
129        }
130    }
131}

133pub fn get_or_load_scripts(fighter: &str, weapon_name: Option<&str>) -> Arc<Vec<LoadedScript>> {
134    let agent = if let Some(weapon) = weapon_name {
135        Hash40::new(fighter).concat_str("_").concat_str(weapon)
136    } else {
137        Hash40::new(fighter)
138    };
139
140    let mut loaded_scripts = LOADED_SCRIPTS.write();
141    if let Some(loaded) = loaded_scripts.get(&agent).and_then(|weak| weak.upgrade()) {
142        return loaded;
143    }
144
145    let scripts = get_scripts(fighter, weapon_name);
146    let scripts = Arc::new(scripts);
147    loaded_scripts.insert(agent, Arc::downgrade(&scripts));
148    scripts
149}

162#[no_mangle]
163extern "C" fn smashline_interpreter(
164    agent: &mut L2CAgentBase,
165    script_name: u64,
166    original: extern "C" fn(&mut L2CAgentBase, &mut Variadic),
167) {
168    let name = Hash40(script_name);
169
170    if let Some(script) =
171        crate::create_agent::user_scripts(agent).and_then(|scripts| scripts.get(&name))
172    {
173        match script {
174            UserScript::Function(func) => {
175                unsafe { (*func)(agent) };
176            }
177            UserScript::Script(script) => {
178                let script = script.read().clone();
179                for action in script.actions.iter() {
180                    if let Err(e) = ACTION_REGISTRY
181                        .execute(action, unsafe { std::mem::transmute(agent as *mut _) })
182                    {
183                        println!("Failed to run action '{}': {e}", action.name);
184                    }
185                }
186            }
187        }
188    } else {
189        if original as *const () == unreachable_smashline_script as *const () {
190            println!(
191                "Unreachable smashline script encountered by {:p}",
192                agent as *mut L2CAgentBase as *mut ()
193            );
194        } else {
195            original(agent, &mut Variadic::new());
196        }
197    }
198}

230static mut CALL_COROUTINE_OFFSET: usize = 0x52e400;
231
232#[skyline::hook(replace = CALL_COROUTINE_OFFSET, inline)]
233unsafe fn call_coroutine_hook(ctx: &mut InlineCtx) {
234    let hash = Hash40(*(ctx.registers[21].x() as *const u64).add(2));
235    ctx.registers[2].set_x(ctx.registers[8].x());
236    ctx.registers[8].set_x(__smashline_interpreter as *const u64 as _);
237    ctx.registers[1].set_x(hash.0);
238}

240static mut CALLING: Option<Hash40> = None;
241#[skyline::hook(offset = 0x372db50, inline)]
242unsafe fn call_function_by_hash(ctx: &InlineCtx) {
243    CALLING = Some(Hash40(ctx.registers[1].x()));
244}
245
246#[skyline::hook(offset = 0x372dcd4, inline)]
247unsafe fn call_by_hash_hook(ctx: &mut InlineCtx) {
248    if let Some(calling) = CALLING.take() {
249        // NOTE: This logic is very fragile, but its the quickest work around I can come up
250        // with to prevent the unreachable script being called unintentionally
251
252        let original = ctx.registers[8].x();
253
254        if original as *const () == unreachable_smashline_script as *const () {
255            ctx.registers[2].set_x(original);
256            ctx.registers[8].set_x(__smashline_interpreter as *const u64 as _);
257            ctx.registers[1].set_x(calling.0);
258        }
259    }
260}

262pub fn install() {
263    skyline::install_hooks!(call_by_hash_hook, call_function_by_hash);
