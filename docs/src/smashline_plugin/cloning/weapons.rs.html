82impl StaticFighterData {
83    pub fn articles_as_slice(&self) -> &[ArticleDescriptor] {
84        unsafe {
85            if self.static_article_info.is_null() {
86                return &[];
87            }
88
89            let ptr = (*self.static_article_info).descriptors;
90            let count = (*self.static_article_info).count;
91            if count == 0 || ptr.is_null() {
92                return &[];
93            }
94
95            std::slice::from_raw_parts(ptr, count)
96        }
97    }
98
99    pub fn get_article(&self, weapon_id: i32) -> Option<ArticleDescriptor> {
100        self.articles_as_slice()
101            .iter()
102            .find(|a| a.weapon_id == weapon_id)
103            .copied()
104    }
105}

132#[skyline::hook(offset = 0x64b730)]
133fn get_static_fighter_data(kind: i32) -> *const StaticFighterData {
134    let original_data: *const StaticFighterData = call_original!(kind);
135
136    if IS_KIRBY_COPYING.load(Ordering::Relaxed) {
137        return original_data;
138    }
139
140    let mut new_descriptors = vec![];
141
142    new_descriptors.extend_from_slice(unsafe { (*original_data).articles_as_slice() });
143
144    for article in new_descriptors.iter_mut() {
145        let weapon_count = WEAPON_COUNT_UPDATE.read();
146        if let Some(new_count) = weapon_count.get(&article.weapon_id) {
147            article.max_count = *new_count;
148        }
149    }
150
151    if let Some(new_articles) = NEW_ARTICLES.read().get(&kind) {
152
153        for article in new_articles.iter() {
154            let source_data = call_original!(article.original_owner);
155
156            unsafe {
157                let Some(article) = (*source_data).get_article(article.weapon_id) else {
158                    panic!("Failed to append article table");
159                };
160
161                new_descriptors.push(article);
162            }
163        }
164    }
165
166    let count = new_descriptors.len();
167    let ptr = new_descriptors.leak().as_ptr();
168    let static_article_info = Box::leak(Box::new(StaticArticleData {
169        descriptors: ptr,
170        count,
171    }));
172
173    let mut new_fighter_data = Box::new(unsafe { *original_data });
174    new_fighter_data.static_article_info = static_article_info as *const StaticArticleData;
175    Box::leak(new_fighter_data)
176}

178fn weapon_owner_hook(ctx: &mut InlineCtx, source_register: usize, dst_register: usize) {
179    if IGNORE_NEW_AGENTS.load(Ordering::Relaxed) {
180        return;
181    }
182
183    let owner = CURRENT_OWNER_KIND.load(Ordering::Relaxed);
184    let agents = NEW_AGENTS.read();
185    let Some(agent) = try_get_new_agent(&agents, unsafe { ctx.registers[source_register].x() as i32 }, owner) else {
186        return;
187    };
188
189    unsafe {
190        ctx.registers[dst_register].set_x(agent.owner_id as u64);
191    }
192}

330unsafe fn kirby_get_copy_articles(ctx: &mut InlineCtx, store_reg: usize) {
331    let kind = CURRENT_KIRBY_COPY.load(Ordering::Relaxed);
332    let kirby_copy_whitelist = KIRBY_COPY_ARTICLE_WHITELIST.read();
333    // println!("Fighter {:#x} is in the whitelist!", kind);
334    let original_descriptors = ctx.registers[store_reg].x() as *const StaticArticleData;
335    IS_KIRBY_COPYING.store(true, Ordering::Relaxed);
336    let fighter_data = get_static_fighter_data(kind);
337    CURRENT_KIRBY_COPY.store(-1, Ordering::Relaxed);
338    IS_KIRBY_COPYING.store(false, Ordering::Relaxed);
339    
340    let mut new_descriptors = vec![];
341    
342    for article in  (*original_descriptors).articles_as_slice().iter() {
343        new_descriptors.push(*article);
344    }
345    
346    for article in new_descriptors.iter_mut() {
347        // println!("checking count for article id {:#x}", article.weapon_id);
348        let weapon_count = WEAPON_COUNT_UPDATE.read();
349        if let Some(new_count) = weapon_count.get(&article.weapon_id) {
350            article.max_count = *new_count;
351        }
352    }
353    
354    if let Some(whitelist) = kirby_copy_whitelist.get(&kind) {
355        for article in (*fighter_data).articles_as_slice().iter() {
356            if whitelist.contains(&article.weapon_id) {
357                // println!("Whitelist contains article {:#x}", article.weapon_id);
358                for descriptor in new_descriptors.iter_mut() {
359                    if article.weapon_id == descriptor.weapon_id {
360                        *descriptor = *article;
361                    }
362                }
363            }
364        }
365    }
366
367    let count = new_descriptors.len();
368    let ptr = new_descriptors.leak().as_ptr();
369    let static_article_info = Box::leak(Box::new(StaticArticleData {
370        descriptors: ptr,
371        count,
372    }));
373
374    ctx.registers[store_reg].set_x(static_article_info as *const StaticArticleData as u64);
375}
